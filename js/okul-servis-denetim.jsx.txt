import React, { useState, useEffect, useCallback } from 'react';
import { Truck, Plus, ChevronLeft, ClipboardCheck, Clock, User, Phone, Hash, Route, AlertCircle, CheckCircle2, XCircle, X, Loader2 } from 'lucide-react';

  { q: 'Aracın yaşı "Okul Servis Araçları Yönetmeliği"nde yer alan yaş şartına uygun mu?', ref: 'OSAY 4/1-f' },
  { q: 'Okul servis aracı temiz, bakımlı, güvenli ve havalandırılmış vaziyette mi?', ref: 'OSAY 4/1-e' },

  { q: 'Arkada "OKUL TAŞITI" yazılı, uygun reflektif kuşak var mı?', ref: 'OSAY 4/1-a' },

const emptyVehicle = { plaka: '', modelYili: '', telefon: '', ogrenciSayisi: '', suruculAdi: '', tcKimlik: '', belgeYili: '', belgeSinifi: '', guzergah: '', gelenAracSayisi: '' };

function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2, 7); }

export default function App() {
  const [vehicles, setVehicles] = useState(null);
  const [inspections, setInspections] = useState({});
  const [view, setView] = useState('list');
  const [activeId, setActiveId] = useState(null);
  const [showAdd, setShowAdd] = useState(false);
  const [form, setForm] = useState(emptyVehicle);
  const [saving, setSaving] = useState(false);
  const [insDraft, setInsDraft] = useState(null);
  const [err, setErr] = useState('');

  useEffect(() => {
    (async () => {
      try {
        const res = await window.storage.get('vehicles', false);
        setVehicles(res ? JSON.parse(res.value) : []);
      } catch { setVehicles([]); }
    })();
  }, []);

  const loadInspections = useCallback(async (vehicleId) => {
    try {
      const res = await window.storage.get(`inspections:${vehicleId}`, false);
      setInspections(prev => ({ ...prev, [vehicleId]: res ? JSON.parse(res.value) : [] }));
    } catch {
      setInspections(prev => ({ ...prev, [vehicleId]: [] }));
    }
  }, []);

  const saveVehicles = async (list) => {
    setVehicles(list);
    try { await window.storage.set('vehicles', JSON.stringify(list), false); }
    catch { setErr('Kaydedilemedi, tekrar deneyin.'); }
  };

  const addVehicle = async () => {
    if (!form.plaka.trim() || !form.suruculAdi.trim()) { setErr('Plaka ve şoför adı zorunlu.'); return; }
    setSaving(true);
    const v = { id: uid(), ...form };
    await saveVehicles([...(vehicles || []), v]);
    setForm(emptyVehicle);
    setShowAdd(false);
    setSaving(false);
    setErr('');
  };

  const openVehicle = (id) => {
    setActiveId(id);
    setView('detail');
    if (!inspections[id]) loadInspections(id);
  };

  const startInspection = () => {
    setInsDraft({
      tarih: new Date().toISOString().slice(0, 10),
      denetleyen: '',
      answers: CHECKLIST.map(() => ({ cevap: null, aciklama: '' })),
    });
    setView('inspect');
  };

  const setAnswer = (i, cevap) => {
    setInsDraft(d => {
      const answers = [...d.answers];
      answers[i] = { ...answers[i], cevap };
      return { ...d, answers };
    });
  };
  const setNote = (i, aciklama) => {
    setInsDraft(d => {
      const answers = [...d.answers];
      answers[i] = { ...answers[i], aciklama };
      return { ...d, answers };
    });
  };

  const saveInspection = async () => {
    if (!insDraft.denetleyen.trim()) { setErr('Denetleyen adı zorunlu.'); return; }
    if (insDraft.answers.some(a => a.cevap === null)) { setErr('Tüm maddeler işaretlenmeli.'); return; }
    setSaving(true);
    const record = { id: uid(), ...insDraft };
    const list = [record, ...(inspections[activeId] || [])];
    try {
      await window.storage.set(`inspections:${activeId}`, JSON.stringify(list), false);
      setInspections(prev => ({ ...prev, [activeId]: list }));
      setInsDraft(null);
      setView('detail');
      setErr('');
    } catch { setErr('Denetim kaydedilemedi, tekrar deneyin.'); }
    setSaving(false);
  };

  const vehicle = vehicles?.find(v => v.id === activeId);

  if (vehicles === null) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-[#F5F3EE]">
        <Loader2 className="animate-spin text-[#1B3A5C]" size={28} />
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#F5F3EE] text-[#1B2430]">
      <header className="bg-[#1B3A5C] text-white px-5 py-4 sticky top-0 z-10 shadow-md">
        <div className="flex items-center gap-3 max-w-3xl mx-auto">
          {view !== 'list' && (
            <button onClick={() => { setView(view === 'inspect' ? 'detail' : 'list'); setErr(''); }} className="p-1 -ml-1 rounded hover:bg-white/10">
              <ChevronLeft size={22} />
            </button>
          )}
          <ClipboardCheck size={22} className="text-[#D4A73B] shrink-0" />
          <div className="min-w-0">
            <h1 className="font-semibold text-[15px] leading-tight truncate">
              {view === 'list' && 'Okul Servis Denetim Takibi'}
              {view === 'detail' && vehicle?.plaka}
              {view === 'inspect' && `Yeni Denetim — ${vehicle?.plaka}`}
            </h1>
            <p className="text-[11px] text-[#B9C6D6] leading-tight">Taşıma Yoluyla Eğitime Erişim Yönetmeliği kapsamı</p>
          </div>
        </div>
      </header>

      <main className="max-w-3xl mx-auto px-4 py-5 pb-24">
        {err && (
          <div className="mb-4 flex items-center gap-2 bg-[#FBEAEA] text-[#9B2C2C] text-sm px-3 py-2 rounded-lg border border-[#F0C6C6]">
            <AlertCircle size={16} className="shrink-0" /> {err}
          </div>
        )}

        {view === 'list' && (
          <ListView vehicles={vehicles} onOpen={openVehicle} onAdd={() => setShowAdd(true)} />
        )}

        {view === 'detail' && vehicle && (
          <DetailView
            vehicle={vehicle}
            records={inspections[activeId] || []}
            onStart={startInspection}
          />
        )}

        {view === 'inspect' && insDraft && (
          <InspectView
            draft={insDraft}
            setAnswer={setAnswer}
            setNote={setNote}
            setField={(k, v) => setInsDraft(d => ({ ...d, [k]: v }))}
            onSave={saveInspection}
            saving={saving}
          />
        )}
      </main>

      {view === 'list' && (
        <button
          onClick={() => setShowAdd(true)}
          className="fixed bottom-6 right-6 bg-[#D4A73B] text-[#1B2430] rounded-full p-4 shadow-lg hover:brightness-105 active:scale-95 transition"
          aria-label="Araç ekle"
        >
          <Plus size={22} />
        </button>
      )}

      {showAdd && (
        <AddVehicleModal
          form={form}
          setForm={setForm}
          onClose={() => { setShowAdd(false); setErr(''); }}
          onSave={addVehicle}
          saving={saving}
        />
      )}
    </div>
  );
}

function ListView({ vehicles, onOpen, onAdd }) {
  if (vehicles.length === 0) {
    return (
      <div className="text-center py-16">
        <Truck size={36} className="mx-auto text-[#B9C6D6] mb-3" />
        <p className="text-[#5A6472] text-sm mb-4">Henüz araç eklenmedi.</p>
        <button onClick={onAdd} className="inline-flex items-center gap-2 bg-[#1B3A5C] text-white px-4 py-2 rounded-lg text-sm font-medium hover:brightness-110">
          <Plus size={16} /> İlk aracı ekle
        </button>
      </div>
    );
  }
  return (
    <div className="space-y-3">
      {vehicles.map(v => (
        <button
          key={v.id}
          onClick={() => onOpen(v.id)}
          className="w-full text-left bg-white rounded-xl p-4 border border-[#E4DFD3] hover:border-[#D4A73B] transition shadow-sm flex items-center gap-3"
        >
          <div className="bg-[#1B3A5C] text-white rounded-lg p-2.5 shrink-0">
            <Truck size={18} />
          </div>
          <div className="min-w-0 flex-1">
            <div className="font-semibold text-[15px] tracking-tight">{v.plaka}</div>
            <div className="text-[13px] text-[#5A6472] truncate">{v.suruculAdi} · {v.guzergah || 'Güzergah belirtilmemiş'}</div>
          </div>
          <div className="text-right text-[11px] text-[#8892A0] shrink-0">
            {v.ogrenciSayisi ? `${v.ogrenciSayisi} öğrenci` : ''}
          </div>
        </button>
      ))}
    </div>
  );
}

function InfoRow({ icon: Icon, label, value }) {
  if (!value) return null;
  return (
    <div className="flex items-center gap-2.5 py-1.5">
      <Icon size={15} className="text-[#8892A0] shrink-0" />
      <span className="text-[12px] text-[#8892A0] w-28 shrink-0">{label}</span>
      <span className="text-[13px] font-medium">{value}</span>
    </div>
  );
}

function DetailView({ vehicle, records, onStart }) {
  return (
    <div>
      <div className="bg-white rounded-xl border border-[#E4DFD3] p-4 mb-5 shadow-sm">
        <InfoRow icon={Hash} label="Model Yılı" value={vehicle.modelYili} />
        <InfoRow icon={User} label="Şoför" value={vehicle.suruculAdi} />
        <InfoRow icon={Phone} label="Telefon" value={vehicle.telefon} />
        <InfoRow icon={Route} label="Güzergah" value={vehicle.guzergah} />
        <InfoRow icon={ClipboardCheck} label="Sürücü Belgesi" value={vehicle.belgeYili && `${vehicle.belgeYili} — ${vehicle.belgeSinifi}`} />
        <InfoRow icon={Truck} label="Öğrenci Sayısı" value={vehicle.ogrenciSayisi} />
      </div>

      <button
        onClick={onStart}
        className="w-full mb-6 bg-[#1B3A5C] text-white py-3 rounded-lg font-medium text-sm flex items-center justify-center gap-2 hover:brightness-110"
      >
        <Plus size={16} /> Yeni Denetim Başlat
      </button>

      <h2 className="text-[13px] font-semibold text-[#5A6472] uppercase tracking-wide mb-2">Denetim Geçmişi</h2>
      {records.length === 0 && <p className="text-[13px] text-[#8892A0]">Henüz denetim kaydı yok.</p>}
      <div className="space-y-2">
        {records.map(r => {
          const hayirSayisi = r.answers.filter(a => a.cevap === (a.invert ? 'evet' : 'hayır')).length;
          const sorunlu = r.answers.reduce((n, a, i) => n + ((CHECKLIST[i].invert ? a.cevap === 'evet' : a.cevap === 'hayır') ? 1 : 0), 0);
          return (
            <div key={r.id} className="bg-white rounded-lg border border-[#E4DFD3] p-3 flex items-center gap-3">
              <div className={`rounded-full p-2 shrink-0 ${sorunlu > 0 ? 'bg-[#FBEAEA] text-[#9B2C2C]' : 'bg-[#EAF5EE] text-[#2F7D4F]'}`}>
                {sorunlu > 0 ? <XCircle size={16} /> : <CheckCircle2 size={16} />}
              </div>
              <div className="min-w-0 flex-1">
                <div className="text-[13px] font-medium">{r.tarih}</div>
                <div className="text-[12px] text-[#8892A0] flex items-center gap-1"><User size={11} /> {r.denetleyen}</div>
              </div>
              <div className="text-[11px] text-right shrink-0 text-[#5A6472]">
                {sorunlu > 0 ? `${sorunlu} madde uygunsuz` : 'Tüm maddeler uygun'}
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
}

function InspectView({ draft, setAnswer, setNote, setField, onSave, saving }) {
  return (
    <div>
      <div className="bg-white rounded-xl border border-[#E4DFD3] p-4 mb-4 shadow-sm space-y-3">
        <div>
          <label className="text-[11px] text-[#8892A0] block mb-1">Denetleyen</label>
          <input
            value={draft.denetleyen}
            onChange={e => setField('denetleyen', e.target.value)}
            placeholder="Ad Soyad / Görev"
            className="w-full border border-[#E4DFD3] rounded-lg px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-[#D4A73B]"
          />
        </div>
        <div>
          <label className="text-[11px] text-[#8892A0] block mb-1">Denetim Tarihi</label>
          <input
            type="date"
            value={draft.tarih}
            onChange={e => setField('tarih', e.target.value)}
            className="w-full border border-[#E4DFD3] rounded-lg px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-[#D4A73B]"
          />
        </div>
      </div>

      <div className="space-y-3">
        {CHECKLIST.map((item, i) => {
          const a = draft.answers[i];
          return (
            <div key={i} className="bg-white rounded-xl border border-[#E4DFD3] p-3.5 shadow-sm">
              <p className="text-[13px] font-medium leading-snug mb-1">{item.q}</p>
              <p className="text-[11px] text-[#8892A0] mb-2.5">{item.ref}</p>
              <div className="flex gap-2 mb-2">
                <button
                  onClick={() => setAnswer(i, 'evet')}
                  className={`flex-1 py-1.5 rounded-lg text-[13px] font-medium border transition ${a.cevap === 'evet' ? 'bg-[#EAF5EE] border-[#2F7D4F] text-[#2F7D4F]' : 'border-[#E4DFD3] text-[#8892A0]'}`}
                >Evet</button>
                <button
                  onClick={() => setAnswer(i, 'hayır')}
                  className={`flex-1 py-1.5 rounded-lg text-[13px] font-medium border transition ${a.cevap === 'hayır' ? 'bg-[#FBEAEA] border-[#9B2C2C] text-[#9B2C2C]' : 'border-[#E4DFD3] text-[#8892A0]'}`}
                >Hayır</button>
              </div>
              <input
                value={a.aciklama}
                onChange={e => setNote(i, e.target.value)}
                placeholder="Açıklama (opsiyonel)"
                className="w-full border border-[#E4DFD3] rounded-lg px-3 py-1.5 text-[12px] focus:outline-none focus:ring-2 focus:ring-[#D4A73B]"
              />
            </div>
          );
        })}
      </div>

      <button
        onClick={onSave}
        disabled={saving}
        className="w-full mt-5 bg-[#1B3A5C] text-white py-3 rounded-lg font-medium text-sm flex items-center justify-center gap-2 hover:brightness-110 disabled:opacity-60"
      >
        {saving ? <Loader2 size={16} className="animate-spin" /> : <ClipboardCheck size={16} />}
        Denetimi Kaydet
      </button>
    </div>
  );
}

function AddVehicleModal({ form, setForm, onClose, onSave, saving }) {
  const set = (k) => (e) => setForm(f => ({ ...f, [k]: e.target.value }));
  const fields = [
    ['plaka', 'Plaka *', 'örn. 23 ACL 183'],
    ['modelYili', 'Model Yılı', 'örn. 2013'],
    ['suruculAdi', 'Şoför Adı Soyadı *', ''],
    ['tcKimlik', 'T.C. Kimlik No', ''],
    ['telefon', 'Telefon', ''],
    ['belgeYili', 'Sürücü Belgesi Yılı', ''],
    ['belgeSinifi', 'Sürücü Belgesi Sınıfı', 'örn. B, C, D'],
    ['guzergah', 'Güzergah', ''],
    ['ogrenciSayisi', 'Öğrenci Sayısı', ''],
    ['gelenAracSayisi', 'Okula Gelen Araç Sayısı', ''],
  ];
  return (
    <div className="fixed inset-0 bg-black/40 flex items-end sm:items-center justify-center z-20 p-0 sm:p-4">
      <div className="bg-white rounded-t-2xl sm:rounded-2xl w-full sm:max-w-md max-h-[85vh] overflow-y-auto">
        <div className="flex items-center justify-between px-4 py-3 border-b border-[#E4DFD3] sticky top-0 bg-white">
          <h3 className="font-semibold text-[15px]">Yeni Araç Ekle</h3>
          <button onClick={onClose} className="p-1 rounded hover:bg-[#F5F3EE]"><X size={18} /></button>
        </div>
        <div className="p-4 space-y-3">
          {fields.map(([k, label, ph]) => (
            <div key={k}>
              <label className="text-[11px] text-[#8892A0] block mb-1">{label}</label>
              <input
                value={form[k]}
                onChange={set(k)}
                placeholder={ph}
                className="w-full border border-[#E4DFD3] rounded-lg px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-[#D4A73B]"
              />
            </div>
          ))}
        </div>
        <div className="p-4 pt-0">
          <button
            onClick={onSave}
            disabled={saving}
            className="w-full bg-[#1B3A5C] text-white py-3 rounded-lg font-medium text-sm flex items-center justify-center gap-2 hover:brightness-110 disabled:opacity-60"
          >
            {saving ? <Loader2 size={16} className="animate-spin" /> : <Plus size={16} />}
            Aracı Kaydet
          </button>
        </div>
      </div>
    </div>
  );
}
